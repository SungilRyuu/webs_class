2. 구구단 출력하기

2.1 for문을 이용하여 2단부터 9단까지 출력하시오!

{
    for(let i = 2; i <= 9; i++){
        document.write("<br>");
        
        for (let j = 1; j <= 9; j++){
            let sum = i * j;
            document.write(i + " * " + j + " = " + sum, "<br>");
        }
    }
}

2.2 while문을 이용하여 2단부터 9단까지 출력하시오!

{
    let i = 2;
    let j = 1;
            
    while (i <= 9) {
        while (j <= 9) {
            let sum = i * j;
            document.write(i + " * " + j + " = " + sum, "<br>");
            j++;
            }
        document.write("<br>");
        i++;
        j = 1;
    }
}
