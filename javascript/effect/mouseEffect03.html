const cursor = document.querySelector(".cursor");

function follow(e){
    const circle1 = document.querySelector(".cursor").clientWidth; //190 -> border 값 제외
    const circle2 = document.querySelector(".cursor").offsetWidth; //200 -> border 값 포함
    const circle3 = document.querySelector(".cursor").getBoundingClientRect(); //200 -> border 값 포함
    
    //getBoundingClientRect: bottom, height, left , right, top, width, x, y

    // console.log(circle1);
    // console.log(circle2);
    // console.log(circle3);

    gsap.to(cursor, {duration: .3, left: e.pageX - (circle3.width/2), top: e.pageY - (circle3.height/2)});

    //출력용
    document.querySelector(".pageX").innerText = e.pageX;
    document.querySelector(".pageY").innerText = e.pageY;
}

window.addEventListener("mousemove", follow);
